Do bro, còn giả định bảng database make_market và JWT_SECRET nên tôi sẽ gửi thêm để bro phân tích và thực hiện chính xác nhất:
1. bảng make_market được thực hiện theo lệnh:
CREATE TABLE make_market (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT UNSIGNED NOT NULL,
    public_key VARCHAR(255) NOT NULL,
    process_name VARCHAR(255) NOT NULL,
    private_key TEXT NOT NULL,
    token_mint VARCHAR(44) NOT NULL,
    sol_amount DECIMAL(18,9) NOT NULL,
    slippage DECIMAL(5,2) NOT NULL,
    delay_seconds INT NOT NULL,
    loop_count INT NOT NULL,
    status ENUM('pending', 'success', 'failed') DEFAULT 'pending',
    buy_tx_id VARCHAR(255) DEFAULT NULL,
    sell_tx_id VARCHAR(255) DEFAULT NULL,
    error TEXT DEFAULT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES accounts(id)
) ENGINE=InnoDB;
